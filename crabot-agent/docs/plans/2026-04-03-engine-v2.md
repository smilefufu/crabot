# Agent Engine V2 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Replace claude-agent-sdk dependency with a self-built execution engine, enabling multi-LLM-format support while keeping the orchestration layer untouched.

**Architecture:** Build `src/engine/` as a self-contained query loop that streams LLM responses, detects tool_use blocks, dispatches tools (parallel for read-only, serial for writes), manages context window via auto-compact, and supports mid-execution human message injection. The engine talks to LLMs via an adapter pattern (Anthropic/OpenAI formats). Worker handler calls the engine instead of SDK's `query()`. MCP servers migrate from claude-agent-sdk's `createSdkMcpServer` to standard `@modelcontextprotocol/sdk`.

**Tech Stack:** TypeScript (ES2020/CommonJS), `@anthropic-ai/sdk`, `@modelcontextprotocol/sdk`, `zod/v4`, vitest

```typescript
// src/engine/types.ts
import { randomUUID } from 'crypto'

// ─── Content Blocks ───

export interface TextBlock {
  type: 'text'
  text: string
}

export interface ImageBlock {
  type: 'image'
  source: {
    type: 'base64' | 'url'
    media_type: string
    data: string
  }
}

export interface ToolUseBlock {
  type: 'tool_use'
  id: string
  name: string
  input: Record<string, unknown>
}

export interface ToolResultBlock {
  type: 'tool_result'
  tool_use_id: string
  content: string
  is_error: boolean
}

export type ContentBlock = TextBlock | ImageBlock | ToolUseBlock | ToolResultBlock

// ─── Messages ───

export interface EngineUserMessage {
  id: string
  role: 'user'
  content: string | ContentBlock[]
  timestamp: string
}

export interface EngineAssistantMessage {
  id: string
  role: 'assistant'
  content: ContentBlock[]
  stopReason: 'end_turn' | 'tool_use' | 'max_tokens' | 'stop_sequence' | null
  timestamp: string
  usage?: { input_tokens: number; output_tokens: number }
}

export interface EngineToolResultMessage {
  id: string
  role: 'user'
  toolResults: Array<{
    tool_use_id: string
    content: string
    is_error: boolean
  }>
  timestamp: string
}

export type EngineMessage = EngineUserMessage | EngineAssistantMessage | EngineToolResultMessage

// ─── Tool Definition ───

export interface ToolDefinition {
  name: string
  description: string
  inputSchema: Record<string, unknown>
  isReadOnly: boolean
  call: (input: Record<string, unknown>, context: ToolCallContext) => Promise<ToolCallResult>
}

export interface ToolCallContext {
  abortSignal?: AbortSignal
  onProgress?: (data: unknown) => void
}

export interface ToolCallResult {
  output: string
  isError: boolean
}

// ─── Stream Chunks ───

export type StreamChunk =
  | { type: 'text_delta'; text: string }
  | { type: 'tool_use_start'; id: string; name: string }
  | { type: 'tool_use_delta'; id: string; partialJson: string }
  | { type: 'tool_use_end'; id: string }
  | { type: 'message_start'; messageId: string }
  | { type: 'message_end'; stopReason: string; usage?: { input_tokens: number; output_tokens: number } }
  | { type: 'error'; error: string }

// ─── Engine Options & Result ───

export interface EngineOptions {
  systemPrompt: string
  tools: ToolDefinition[]
  model: string
  maxTurns?: number
  maxTokens?: number
  abortSignal?: AbortSignal
  /** Callback for each completed turn (for progress reporting) */
  onTurn?: (turn: EngineTurnEvent) => void
  /** Callback for streaming text (for real-time output) */
  onTextDelta?: (text: string) => void
  /** Inject human messages mid-execution */
  humanMessageQueue?: AsyncIterable<string>
}

export interface EngineTurnEvent {
  turnNumber: number
  assistantText: string
  toolCalls: Array<{ name: string; input: Record<string, unknown>; output: string; isError: boolean }>
  stopReason: string | null
}

export interface EngineResult {
  outcome: 'completed' | 'failed' | 'max_turns' | 'aborted'
  finalText: string
  totalTurns: number
  usage: { totalInputTokens: number; totalOutputTokens: number }
  error?: string
}

// ─── Factories ───

export function createUserMessage(content: string | ContentBlock[]): EngineUserMessage {
  return {
    id: randomUUID(),
    role: 'user',
    content,
    timestamp: new Date().toISOString(),
  }
}

export function createAssistantMessage(
  content: ContentBlock[],
  stopReason: EngineAssistantMessage['stopReason'],
  usage?: { input_tokens: number; output_tokens: number },
): EngineAssistantMessage {
  return {
    id: randomUUID(),
    role: 'assistant',
    content,
    stopReason,
    timestamp: new Date().toISOString(),
    usage,
  }
}

export function createToolResultMessage(
  toolUseId: string,
  content: string,
  isError: boolean,
): EngineToolResultMessage {
  return {
    id: randomUUID(),
    role: 'user',
    toolResults: [{ tool_use_id: toolUseId, content, is_error: isError }],
    timestamp: new Date().toISOString(),
  }
}
```

```typescript
// src/engine/llm-adapter.ts
import Anthropic from '@anthropic-ai/sdk'
import type {
  MessageParam,
  ContentBlockParam,
  Tool as AnthropicTool,
} from '@anthropic-ai/sdk/resources/messages'
import type {
  EngineMessage,
  StreamChunk,
  ToolDefinition,
  ContentBlock,
} from './types.js'

// ─── Adapter Interface ───

export interface LLMStreamParams {
  messages: EngineMessage[]
  systemPrompt: string
  tools: ToolDefinition[]
  model: string
  maxTokens?: number
  signal?: AbortSignal
}

export interface LLMAdapter {
  stream(params: LLMStreamParams): AsyncGenerator<StreamChunk>
  updateConfig(config: Partial<LLMAdapterConfig>): void
}

export interface LLMAdapterConfig {
  endpoint: string
  apikey: string
  model?: string
  maxTokens?: number
}

// ─── Message Normalization ───

export function normalizeMessagesForAnthropic(messages: EngineMessage[]): MessageParam[] {
  return messages.map((msg): MessageParam => {
    if (msg.role === 'assistant') {
      return {
        role: 'assistant',
        content: msg.content.map(toAnthropicContentBlock),
      }
    }

    // EngineToolResultMessage
    if ('toolResults' in msg) {
      return {
        role: 'user',
        content: msg.toolResults.map((tr) => ({
          type: 'tool_result' as const,
          tool_use_id: tr.tool_use_id,
          content: tr.content,
          is_error: tr.is_error,
        })),
      }
    }

    // EngineUserMessage
    if (typeof msg.content === 'string') {
      return { role: 'user', content: msg.content }
    }

    return {
      role: 'user',
      content: msg.content.map(toAnthropicContentBlock),
    }
  })
}

function toAnthropicContentBlock(block: ContentBlock): ContentBlockParam {
  switch (block.type) {
    case 'text':
      return { type: 'text', text: block.text }
    case 'image':
      return {
        type: 'image',
        source: {
          type: block.source.type as 'base64',
          media_type: block.source.media_type as 'image/png',
          data: block.source.data,
        },
      }
    case 'tool_use':
      return {
        type: 'tool_use',
        id: block.id,
        name: block.name,
        input: block.input,
      }
    case 'tool_result':
      return {
        type: 'tool_result',
        tool_use_id: block.tool_use_id,
        content: block.content,
        is_error: block.is_error,
      }
  }
}

// ─── Anthropic Adapter ───

export class AnthropicAdapter implements LLMAdapter {
  private client: Anthropic
  private config: LLMAdapterConfig

  constructor(config: LLMAdapterConfig) {
    this.config = config
    this.client = new Anthropic({
      baseURL: config.endpoint,
      apiKey: config.apikey,
    })
  }

  static toAnthropicTool(tool: ToolDefinition): AnthropicTool {
    return {
      name: tool.name,
      description: tool.description,
      input_schema: tool.inputSchema as AnthropicTool['input_schema'],
    }
  }

  updateConfig(config: Partial<LLMAdapterConfig>): void {
    this.config = { ...this.config, ...config }
    if (config.endpoint || config.apikey) {
      this.client = new Anthropic({
        baseURL: this.config.endpoint,
        apiKey: this.config.apikey,
      })
    }
  }

  async *stream(params: LLMStreamParams): AsyncGenerator<StreamChunk> {
    const anthropicMessages = normalizeMessagesForAnthropic(params.messages)
    const anthropicTools = params.tools.map(AnthropicAdapter.toAnthropicTool)

    const streamResponse = this.client.messages.stream({
      model: params.model,
      max_tokens: params.maxTokens ?? 16384,
      system: params.systemPrompt,
      messages: anthropicMessages,
      ...(anthropicTools.length > 0 ? { tools: anthropicTools } : {}),
    }, {
      signal: params.signal ?? undefined,
    })

    try {
      for await (const event of streamResponse) {
        const chunks = mapAnthropicEvent(event)
        for (const chunk of chunks) {
          yield chunk
        }
      }
    } finally {
      streamResponse.controller.abort()
    }
  }
}

function mapAnthropicEvent(event: Anthropic.MessageStreamEvent): StreamChunk[] {
  switch (event.type) {
    case 'message_start':
      return [{ type: 'message_start', messageId: event.message.id }]

    case 'content_block_start':
      if (event.content_block.type === 'tool_use') {
        return [{
          type: 'tool_use_start',
          id: event.content_block.id,
          name: event.content_block.name,
        }]
      }
      return []

    case 'content_block_delta':
      if (event.delta.type === 'text_delta') {
        return [{ type: 'text_delta', text: event.delta.text }]
      }
      if (event.delta.type === 'input_json_delta') {
        return [{
          type: 'tool_use_delta',
          id: '', // Anthropic doesn't send id in delta, track via index
          partialJson: event.delta.partial_json,
        }]
      }
      return []

    case 'content_block_stop':
      return [{ type: 'tool_use_end', id: '' }]

    case 'message_delta':
      return [{
        type: 'message_end',
        stopReason: event.delta.stop_reason ?? 'end_turn',
        usage: event.usage ? {
          input_tokens: 0, // message_delta only has output_tokens
          output_tokens: event.usage.output_tokens,
        } : undefined,
      }]

    case 'message_stop':
      return []

    default:
      return []
  }
}
```

```typescript
// src/engine/stream-processor.ts
import { jsonrepair } from 'jsonrepair'
import type { StreamChunk, ToolUseBlock } from './types.js'

export interface ProcessedResponse {
  text: string
  toolUseBlocks: ToolUseBlock[]
  stopReason: string | null
  usage?: { input_tokens: number; output_tokens: number }
}

export class StreamProcessor {
  private text = ''
  private currentToolId: string | null = null
  private currentToolName: string | null = null
  private currentToolJsonParts: string[] = []
  private toolUseBlocks: ToolUseBlock[] = []
  private stopReason: string | null = null
  private usage?: { input_tokens: number; output_tokens: number }

  process(chunk: StreamChunk): void {
    switch (chunk.type) {
      case 'text_delta':
        this.text += chunk.text
        break

      case 'tool_use_start':
        this.currentToolId = chunk.id
        this.currentToolName = chunk.name
        this.currentToolJsonParts = []
        break

      case 'tool_use_delta':
        this.currentToolJsonParts.push(chunk.partialJson)
        break

      case 'tool_use_end':
        if (this.currentToolId && this.currentToolName) {
          const rawJson = this.currentToolJsonParts.join('')
          const input = this.parseToolInput(rawJson)
          this.toolUseBlocks.push({
            type: 'tool_use',
            id: this.currentToolId,
            name: this.currentToolName,
            input,
          })
        }
        this.currentToolId = null
        this.currentToolName = null
        this.currentToolJsonParts = []
        break

      case 'message_end':
        this.stopReason = chunk.stopReason
        if (chunk.usage) {
          this.usage = chunk.usage
        }
        break
    }
  }

  finalize(): ProcessedResponse {
    return {
      text: this.text,
      toolUseBlocks: [...this.toolUseBlocks],
      stopReason: this.stopReason,
      usage: this.usage,
    }
  }

  reset(): void {
    this.text = ''
    this.currentToolId = null
    this.currentToolName = null
    this.currentToolJsonParts = []
    this.toolUseBlocks = []
    this.stopReason = null
    this.usage = undefined
  }

  private parseToolInput(raw: string): Record<string, unknown> {
    if (!raw || raw.trim() === '') return {}
    try {
      return JSON.parse(raw)
    } catch {
      try {
        return JSON.parse(jsonrepair(raw))
      } catch {
        return { _raw: raw }
      }
    }
  }
}
```

```typescript
// src/engine/tool-framework.ts
import type { ToolDefinition, ToolUseBlock, ToolCallContext, ToolCallResult } from './types.js'

// ─── Tool Definition Helper ───

export interface DefineToolParams {
  name: string
  description: string
  inputSchema: Record<string, unknown>
  isReadOnly?: boolean
  call: (input: Record<string, unknown>, context: ToolCallContext) => Promise<ToolCallResult>
}

export function defineTool(params: DefineToolParams): ToolDefinition {
  return {
    name: params.name,
    description: params.description,
    inputSchema: params.inputSchema,
    isReadOnly: params.isReadOnly ?? false,
    call: params.call,
  }
}

// ─── Tool Lookup ───

export function findTool(tools: ToolDefinition[], name: string): ToolDefinition | undefined {
  return tools.find((t) => t.name === name)
}

// ─── Tool Call Partitioning ───

export interface ToolBatch {
  parallel: boolean
  blocks: ToolUseBlock[]
}

export function partitionToolCalls(
  blocks: ToolUseBlock[],
  tools: ToolDefinition[],
): ToolBatch[] {
  if (blocks.length === 0) return []

  const batches: ToolBatch[] = []
  let currentBatch: ToolUseBlock[] = []
  let currentIsReadOnly: boolean | null = null

  for (const block of blocks) {
    const tool = findTool(tools, block.name)
    const isReadOnly = tool?.isReadOnly ?? false

    if (currentIsReadOnly === null) {
      currentIsReadOnly = isReadOnly
      currentBatch.push(block)
      continue
    }

    if (isReadOnly && currentIsReadOnly) {
      // Both read-only — extend parallel batch
      currentBatch.push(block)
    } else {
      // Flush current batch
      batches.push({
        parallel: currentIsReadOnly && currentBatch.length > 0,
        blocks: currentBatch,
      })
      currentBatch = [block]
      currentIsReadOnly = isReadOnly
    }
  }

  // Flush final batch
  if (currentBatch.length > 0) {
    batches.push({
      parallel: currentIsReadOnly === true,
      blocks: currentBatch,
    })
  }

  return batches
}
```

```typescript
// src/engine/tool-orchestration.ts
import type { ToolDefinition, ToolCallContext } from './types.js'
import { findTool, type ToolBatch } from './tool-framework.js'

export interface ToolResultEntry {
  tool_use_id: string
  content: string
  is_error: boolean
}

const MAX_CONCURRENCY = 10

export async function executeToolBatches(
  batches: ToolBatch[],
  tools: ToolDefinition[],
  context?: ToolCallContext,
): Promise<ToolResultEntry[]> {
  const allResults: ToolResultEntry[] = []

  for (const batch of batches) {
    if (batch.parallel) {
      const results = await executeParallel(batch.blocks, tools, context)
      allResults.push(...results)
    } else {
      const results = await executeSerial(batch.blocks, tools, context)
      allResults.push(...results)
    }
  }

  return allResults
}

async function executeParallel(
  blocks: Array<{ id: string; name: string; input: Record<string, unknown> }>,
  tools: ToolDefinition[],
  context?: ToolCallContext,
): Promise<ToolResultEntry[]> {
  // Execute in chunks of MAX_CONCURRENCY
  const results: ToolResultEntry[] = new Array(blocks.length)

  for (let i = 0; i < blocks.length; i += MAX_CONCURRENCY) {
    const chunk = blocks.slice(i, i + MAX_CONCURRENCY)
    const chunkResults = await Promise.all(
      chunk.map((block, idx) => executeSingleTool(block, tools, context).then(r => ({ idx: i + idx, result: r }))),
    )
    for (const { idx, result } of chunkResults) {
      results[idx] = result
    }
  }

  return results
}

async function executeSerial(
  blocks: Array<{ id: string; name: string; input: Record<string, unknown> }>,
  tools: ToolDefinition[],
  context?: ToolCallContext,
): Promise<ToolResultEntry[]> {
  const results: ToolResultEntry[] = []
  for (const block of blocks) {
    const result = await executeSingleTool(block, tools, context)
    results.push(result)
  }
  return results
}

async function executeSingleTool(
  block: { id: string; name: string; input: Record<string, unknown> },
  tools: ToolDefinition[],
  context?: ToolCallContext,
): Promise<ToolResultEntry> {
  const tool = findTool(tools, block.name)
  if (!tool) {
    return {
      tool_use_id: block.id,
      content: `Tool not found: ${block.name}`,
      is_error: true,
    }
  }

  try {
    const result = await tool.call(block.input, context ?? {})
    return {
      tool_use_id: block.id,
      content: result.output,
      is_error: result.isError,
    }
  } catch (error) {
    const msg = error instanceof Error ? error.message : String(error)
    return {
      tool_use_id: block.id,
      content: `Tool execution error: ${msg}`,
      is_error: true,
    }
  }
}
```

```typescript
// src/engine/context-manager.ts
import { createUserMessage, type EngineMessage } from './types.js'

export interface ContextManagerOptions {
  maxContextTokens: number
  compactThreshold?: number    // 0-1, default 0.8
  keepRecentMessages?: number  // messages to keep after compact, default 6
}

export class ContextManager {
  private maxContextTokens: number
  private compactThreshold: number
  private keepRecentMessages: number
  private totalInputTokens = 0
  private totalOutputTokens = 0

  constructor(options: ContextManagerOptions) {
    this.maxContextTokens = options.maxContextTokens
    this.compactThreshold = options.compactThreshold ?? 0.8
    this.keepRecentMessages = options.keepRecentMessages ?? 6
  }

  /**
   * Rough token estimation: ~4 characters per token + overhead per message
   */
  estimateMessageTokens(msg: EngineMessage): number {
    const MESSAGE_OVERHEAD = 4 // role, separator tokens
    let charCount = 0

    if (msg.role === 'assistant') {
      for (const block of msg.content) {
        switch (block.type) {
          case 'text':
            charCount += block.text.length
            break
          case 'tool_use':
            charCount += block.name.length + JSON.stringify(block.input).length + 20
            break
        }
      }
    } else if ('toolResults' in msg) {
      for (const tr of msg.toolResults) {
        charCount += tr.content.length + 20
      }
    } else {
      // User message
      if (typeof msg.content === 'string') {
        charCount = msg.content.length
      } else {
        for (const block of msg.content) {
          if (block.type === 'text') charCount += block.text.length
          else if (block.type === 'image') charCount += 1000 // rough estimate for images
          else if (block.type === 'tool_use') charCount += JSON.stringify(block.input).length + 20
        }
      }
    }

    return Math.ceil(charCount / 4) + MESSAGE_OVERHEAD
  }

  estimateTotalTokens(messages: EngineMessage[]): number {
    return messages.reduce((sum, msg) => sum + this.estimateMessageTokens(msg), 0)
  }

  shouldCompact(messages: EngineMessage[]): boolean {
    const estimated = this.estimateTotalTokens(messages)
    return estimated >= this.maxContextTokens * this.compactThreshold
  }

  /**
   * Simple compaction: summarize old messages, keep recent ones.
   * Phase 3 will add LLM-based summarization (like claude-code's compact service).
   */
  compactMessages(messages: EngineMessage[]): EngineMessage[] {
    if (messages.length <= this.keepRecentMessages) {
      return messages
    }

    const oldMessages = messages.slice(0, messages.length - this.keepRecentMessages)
    const recentMessages = messages.slice(messages.length - this.keepRecentMessages)

    // Build a simple text summary of old messages
    const summaryParts: string[] = ['[Earlier conversation summary]']
    for (const msg of oldMessages) {
      if (msg.role === 'assistant') {
        const text = msg.content
          .filter((b): b is { type: 'text'; text: string } => b.type === 'text')
          .map((b) => b.text)
          .join(' ')
        if (text) summaryParts.push(`Assistant: ${text.slice(0, 200)}`)
      } else if (msg.role === 'user' && !('toolResults' in msg)) {
        const text = typeof msg.content === 'string'
          ? msg.content
          : msg.content
              .filter((b): b is { type: 'text'; text: string } => b.type === 'text')
              .map((b) => b.text)
              .join(' ')
        if (text) summaryParts.push(`User: ${text.slice(0, 200)}`)
      }
    }

    const summaryMessage = createUserMessage(summaryParts.join('\n'))
    return [summaryMessage, ...recentMessages]
  }

  updateFromUsage(usage: { input_tokens: number; output_tokens: number }): void {
    this.totalInputTokens += usage.input_tokens
    this.totalOutputTokens += usage.output_tokens
  }

  getCumulativeUsage(): { totalInputTokens: number; totalOutputTokens: number } {
```

```typescript
// src/engine/query-loop.ts
import type {
  EngineMessage,
  EngineOptions,
  EngineResult,
  EngineTurnEvent,
  ToolUseBlock,
} from './types.js'
import { createUserMessage, createAssistantMessage, createToolResultMessage } from './types.js'
import type { LLMAdapter } from './llm-adapter.js'
import { StreamProcessor } from './stream-processor.js'
import { partitionToolCalls } from './tool-framework.js'
import { executeToolBatches } from './tool-orchestration.js'
import { ContextManager } from './context-manager.js'

export interface RunEngineParams {
  prompt: string
  adapter: LLMAdapter
  options: EngineOptions
}

export async function runEngine(params: RunEngineParams): Promise<EngineResult> {
  const { prompt, adapter, options } = params
  const {
    systemPrompt,
    tools,
    model,
    maxTurns = 100,
    maxTokens = 16384,
    abortSignal,
    onTurn,
    onTextDelta,
  } = options

  const contextManager = new ContextManager({
    maxContextTokens: 200000, // default, can be made configurable
  })

  const messages: EngineMessage[] = [createUserMessage(prompt)]
  const streamProcessor = new StreamProcessor()
  let totalTurns = 0
  let lastText = ''

  for (let turn = 0; turn < maxTurns; turn++) {
    // Check abort
    if (abortSignal?.aborted) {
      return {
        outcome: 'aborted',
        finalText: lastText,
        totalTurns,
        usage: contextManager.getCumulativeUsage(),
      }
    }

    // Check if context needs compaction
    if (contextManager.shouldCompact(messages)) {
      const compacted = contextManager.compactMessages(messages)
      messages.length = 0
      messages.push(...compacted)
    }

    // Stream LLM response
    streamProcessor.reset()
    const toolCallsForTurn: Array<{
      name: string
      input: Record<string, unknown>
      output: string
      isError: boolean
    }> = []

    try {
      for await (const chunk of adapter.stream({
        messages,
        systemPrompt,
        tools,
        model,
        maxTokens,
        signal: abortSignal,
      })) {
        // Check abort between chunks
        if (abortSignal?.aborted) {
          return {
            outcome: 'aborted',
            finalText: lastText,
            totalTurns,
            usage: contextManager.getCumulativeUsage(),
          }
        }

        streamProcessor.process(chunk)

        // Forward text deltas for streaming output
        if (chunk.type === 'text_delta' && onTextDelta) {
          onTextDelta(chunk.text)
        }
      }
    } catch (error) {
      if (abortSignal?.aborted) {
        return {
          outcome: 'aborted',
          finalText: lastText,
          totalTurns,
          usage: contextManager.getCumulativeUsage(),
        }
      }
      const msg = error instanceof Error ? error.message : String(error)
      return {
        outcome: 'failed',
        finalText: lastText,
        totalTurns,
        usage: contextManager.getCumulativeUsage(),
        error: msg,
      }
    }

    const processed = streamProcessor.finalize()
    totalTurns++

    if (processed.usage) {
      contextManager.updateFromUsage(processed.usage)
    }

    // Build assistant message content
    const assistantContent = [
      ...(processed.text ? [{ type: 'text' as const, text: processed.text }] : []),
      ...processed.toolUseBlocks,
    ]
    if (assistantContent.length > 0) {
      messages.push(
        createAssistantMessage(
          assistantContent,
          processed.stopReason as 'end_turn' | 'tool_use' | 'max_tokens' | null,
          processed.usage,
        ),
      )
    }

    lastText = processed.text || lastText

    // If stop_reason is end_turn or no tool_use → done
    if (processed.stopReason !== 'tool_use' || processed.toolUseBlocks.length === 0) {
      // Report turn
      if (onTurn) {
        onTurn({
          turnNumber: totalTurns,
          assistantText: processed.text,
          toolCalls: toolCallsForTurn,
          stopReason: processed.stopReason,
        })
      }

      return {
        outcome: 'completed',
        finalText: lastText,
        totalTurns,
        usage: contextManager.getCumulativeUsage(),
      }
    }

    // Execute tools
    const batches = partitionToolCalls(processed.toolUseBlocks, tools)
    const toolResults = await executeToolBatches(batches, tools, {
      abortSignal,
    })

    // Track tool calls for turn event
    for (let i = 0; i < processed.toolUseBlocks.length; i++) {
      const block = processed.toolUseBlocks[i]
      const result = toolResults[i]
      toolCallsForTurn.push({
        name: block.name,
        input: block.input,
        output: result?.content ?? '',
        isError: result?.is_error ?? true,
      })
    }

    // Add tool results to messages
    for (const result of toolResults) {
      messages.push(createToolResultMessage(result.tool_use_id, result.content, result.is_error))
    }

    // Report turn
    if (onTurn) {
      onTurn({
        turnNumber: totalTurns,
        assistantText: processed.text,
        toolCalls: toolCallsForTurn,
        stopReason: processed.stopReason,
      })
    }
  }

  // Max turns reached
  return {
    outcome: 'max_turns',
    finalText: lastText,
    totalTurns,
    usage: contextManager.getCumulativeUsage(),
  }
}
```

```typescript
import { McpServer } from '@modelcontextprotocol/sdk/server/mcp.js'
import { z } from 'zod/v4'

export interface McpToolDef<T extends z.ZodType> {
  name: string
  description: string
  schema: T
  handler: (args: z.infer<T>) => Promise<{ content: Array<{ type: 'text'; text: string }> }>
}

export function createMcpServerWithTools(
  config: { name: string; version: string },
  tools: McpToolDef<z.ZodType>[],
): McpServer {
  const server = new McpServer(config)

  for (const toolDef of tools) {
    server.tool(
      toolDef.name,
      toolDef.description,
      Object.fromEntries(
        Object.entries((toolDef.schema as z.ZodObject<z.ZodRawShape>).shape ?? {}).map(
          ([key, val]) => [key, val],
        ),
      ),
      async (args: Record<string, unknown>) => {
        const parsed = toolDef.schema.parse(args)
        return toolDef.handler(parsed)
      },
    )
  }

  return server
}
```
